# PHASE 2: Discover and learn from local project files
# =============================================================================

def scan_local_files(deep: bool = False):
    """Discover readable files in the project and learn from them."""
    print("\n  Phase 2: Scanning local project files...")
    
    # Directories to scan
    scan_dirs = [
        (os.path.join(BASE_DIR, "doc"), "project documentation"),
        (os.path.join(BASE_DIR, "src"), "compression source code"),
        (os.path.join(BASE_DIR, "server"), "server source code"),
    ]
    
    if deep:
        scan_dirs.extend([
            (os.path.join(BASE_DIR, "desktop_app", "src"), "desktop app source"),
            (os.path.join(BASE_DIR, "desktop_app", "electron"), "electron source"),
        ])
    
    learned = 0
    
    for scan_dir, category in scan_dirs:
        if not os.path.exists(scan_dir):
            continue
        
        for root, dirs, files in os.walk(scan_dir):
            # Skip node_modules, dist, build dirs
            dirs[:] = [d for d in dirs if d not in {'node_modules', 'dist', 'build', '.git', '__pycache__', 'dist-electron'}]
            
            for fname in files:
                filepath = os.path.join(root, fname)
                ext = os.path.splitext(fname)[1].lower()
                
                # Skip files we can't read or that are too small
                if not can_read(filepath):
                    continue
                
                try:
                    size = os.path.getsize(filepath)
                except OSError:
                    continue
                    
                if size < 50 or size > 5 * 1024 * 1024:
                    continue
                
                # Extract text
                text = extract_text(filepath)
                if not text or len(text.strip()) < 50:
                    continue
                
                # Generate topic from path
                rel_path = os.path.relpath(filepath, BASE_DIR)
                topic = f"{category} - {fname}"
                
                print(f"  [LOCAL] {rel_path} ({size} bytes)...")
                try:
                    result = store_knowledge(topic, text, source="local_file")
                    tokens = result.get("token_count", 0)
                    savings = result.get("savings_pct", 0)
                    print(f"          -> {tokens} tokens, {savings}% compressed")
                    learned += 1
                except Exception as e:
                    print(f"          -> Error: {e}")
    
    return learned

def learn_from_web(deep: bool = False):
    """Dynamically learn from real web sources."""
    print("\n  Phase 3: Learning from the web (live content)...")
    
    sources = WEB_SOURCES.copy()
    if deep:
        sources.extend(DEEP_WEB_SOURCES)
    
    learned = 0
    failed = 0
    
    for topic, url in sources:
        print(f"  [WEB] {topic}...")
        try:
            result = learn_from_url(topic, url)
            if "error" in result:
                print(f"        -> Failed: {result['error']}")
                failed += 1
            else:
                tokens = result.get("token_count", 0)
                savings = result.get("savings_pct", 0)
                print(f"        -> {tokens} tokens, {savings}% compressed")
                learned += 1
        except Exception as e:
            print(f"        -> Error: {e}")
            failed += 1
        
        # Small delay to be polite to servers
        time.sleep(0.5)
    
    if failed > 0:
        print(f"\n  Note: {failed} sources failed (network issues or blocked). "
              f"The brain learned from {learned} sources successfully.")
    
    return learned

def main():
    parser = argparse.ArgumentParser(description="Neural Brain Dynamic Training")
    parser.add_argument("--local", action="store_true", help="Scan and learn from local project files")
    parser.add_argument("--web", action="store_true", help="Learn from web sources")
    parser.add_argument("--all", action="store_true", help="Both local + web (default)")
    parser.add_argument("--deep", action="store_true", help="Deep mode: more files, more web sources")
    args = parser.parse_args()
    
    # Default to --all if nothing specified
    if not args.local and not args.web and not args.all:
        args.all = True
    
    do_local = args.local or args.all
    do_web = args.web or args.all
    
    ensure_brain()
    
    print("\n  +--------------------------------------------------+")
    print("  |  Neural Brain - Dynamic Training                 |")
    print("  |  Learning from real sources, not static text     |")
    print("  +--------------------------------------------------+")
    
    total_learned = 0
    start_time = time.time()
    
    # Phase 1: Self-knowledge (minimal, just identity)
    print("\n  Phase 1: Core self-knowledge...")
    result = store_knowledge("neural studio identity", SELF_KNOWLEDGE.strip(), source="self")
    print(f"  [SELF] Identity loaded ({result.get('token_count', 0)} tokens)")
    total_learned += 1
    
    # Phase 2: Local files
    if do_local:
        local_count = scan_local_files(deep=args.deep)
        total_learned += local_count
        print(f"\n  Local files learned: {local_count}")
    
    # Phase 3: Web sources
    if do_web:
        web_count = learn_from_web(deep=args.deep)
        total_learned += web_count
        print(f"\n  Web sources learned: {web_count}")
    
    # Summary
    elapsed = time.time() - start_time
    stats = brain_stats()
    
    print("\n  +--------------------------------------------------+")
    print(f"  |  Training Complete in {elapsed:.1f}s")
    print(f"  |  Items learned this session: {total_learned}")
    print(f"  |  Total knowledge items:  {stats['total_knowledge_items']}")
    print(f"  |  Topics:                 {stats['total_topics']}")
    print(f"  |  Vocabulary:             {stats['vocabulary_size']} words")
    print(f"  |  Raw size:               {stats['total_raw_human']}")
    print(f"  |  Compressed:             {stats['total_compressed_human']}")
    print(f"  |  Savings:                {stats['compression_savings_pct']}%")
    print("  +--------------------------------------------------+")
    print(f"\n  The brain is ready. Run `npm run dev` to start.\n")

if __name__ == "__main__":
    main()
