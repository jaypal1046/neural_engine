"""
RLHF Trainer — Anthropic-Style Training for Neural Studio
==========================================================

HOW ANTHROPIC TRAINS CLAUDE (simplified):

  1. SUPERVISED FINE-TUNING (SFT)
     → Show model: (prompt → ideal answer) pairs
     → Model learns the "baseline good behavior"

  2. REWARD MODEL (RM)
     → Humans rank multiple answers: A > B > C
     → Train a "judge" network: score(answer) → 0.0–1.0
     → This judge scores any future answer automatically

  3. REINFORCEMENT LEARNING FROM HUMAN FEEDBACK (RLHF / PPO)
     → Generate answers → Reward Model scores them
     → Adjust model weights to produce higher-scored answers
     → Model improves toward "what humans prefer"

  4. CONSTITUTIONAL AI (CAI)
     → A fixed set of rules (the "constitution")
     → AI critiques its OWN answers against the rules
     → Revises answers until they pass all rules
     → No humans needed — self-improving loop

WE IMPLEMENT ALL 4 for Neural Studio:
  - Reward Model  → scores any AI response 0.0–1.0
  - SFT pairs     → teach ideal (question → answer) examples
  - CAI critique  → AI checks its own answers against rules
  - RLHF feedback → human thumbs up/down improve future answers

Run:
    python rlhf_trainer.py --sft          # Supervised fine-tuning
    python rlhf_trainer.py --feedback     # Process human feedback
    python rlhf_trainer.py --cai          # Run Constitutional AI critique
    python rlhf_trainer.py --full         # Run everything (recommended)
    python rlhf_trainer.py --status       # Show training stats
"""

# STEP 2: REWARD MODEL  — Scores any response 0.0 to 1.0
# =============================================================================

class RewardModel:
    """
    Scores an AI response based on multiple quality signals.
    Higher = better. Range 0.0 to 1.0.
    
    Anthropic trains a neural network for this.
    We use a rule-based ensemble (same spirit, smaller compute).
    """

    def score(self, question: str, answer: str, conversation_context: str = "") -> Dict:
        scores = {}
        q_lower = question.lower()
        a_lower = answer.lower()
        words   = len(answer.split())

        # 1. Relevance — does the answer address the question?
        q_tokens = set(self._tokens(question))
        a_tokens = set(self._tokens(answer))
        if len(q_tokens) <= 3:
            # Short questions (What are you? Hi.) → give benefit of the doubt
            scores["relevance"] = 0.85
        else:
            overlap = len(q_tokens & a_tokens) / max(len(q_tokens), 1)
            scores["relevance"] = min(overlap * 2.5, 1.0)

        # 2. Length quality — Goldilocks zone
        if words < 5:
            scores["length"] = 0.1
        elif words < 10:
            # Acceptable for greeting/simple questions
            is_simple = len(q_tokens) <= 4
            scores["length"] = 0.75 if is_simple else 0.3
        elif words <= 250:
            scores["length"] = 1.0
        elif words <= 450:
            scores["length"] = 0.85
        else:
            scores["length"] = 0.6  # Too verbose

        # 3. Specificity — numbers, code, bold, examples
        specific_hits = sum(1 for p in [
            r'\b\d+[.,]?\d*\b',           # Numbers
            r'e\.g\.|for example|such as', # Examples
            r'`[^`]+`',                    # Inline code
            r'\*\*[^*]+\*\*',             # Bold
            r'\b(specifically|step \d)',   # Specificity markers
            r'^\d+\.\s',                  # Numbered list
            r'^[-•*]\s',                  # Bullet list
        ] if re.search(p, answer, re.M | re.I))
        scores["specificity"] = min(specific_hits * 0.18, 1.0)

        # 4. Honesty — appropriate hedging vs overclaiming
        hedge   = ["i think", "i believe", "likely", "probably", "may ",
                   "might ", "could ", "i don't know", "not sure", "uncertain"]
        overcon = ["definitely", "certainly", "guaranteed", "100%", "absolute"]
        n_hedge   = sum(1 for w in hedge   if w in a_lower)
        n_over    = sum(1 for w in overcon if w in a_lower)
        if n_over > 2:
            scores["honesty"] = 0.3
        elif n_hedge > 0:
            scores["honesty"] = min(0.7 + n_hedge * 0.1, 1.0)
        else:
            scores["honesty"] = 0.65  # Neutral (not overclaiming either)

        # 5. Structure — formatting quality
        sentences   = [s.strip() for s in re.split(r'[.!?]', answer) if s.strip()]
        cap_ratio   = sum(1 for s in sentences if s and s[0].isupper()) / max(len(sentences), 1)
        has_list    = bool(re.search(r'(^[-•*\d]\.?\s)', answer, re.M))
        has_header  = bool(re.search(r'\*\*[^*]+\*\*', answer))
        struct = cap_ratio * 0.5
        if has_list:   struct += 0.3
        if has_header: struct += 0.2
        scores["structure"] = min(struct, 1.0)

        # 6. Unknown handling — honest unknowing is virtuous
        if any(p in a_lower for p in ["don't have knowledge", "don't know", "teach me",
                                       "haven't learned", "no knowledge"]):
            scores["unknown_handling"] = 0.9
        else:
            scores["unknown_handling"] = 0.7

        # 7. Domain knowledge — Neural Studio terminology
        domain_terms = [
            "cmix", "bwt", "lz77", "ppm", "entropy", "compress", "huffman",
            "arithmetic", "context mixing", "advisor", "neural", "brain",
            "knowledge", "token", "tfidf", "tf-idf", "myzip", "aiz",
            "range coding", "predictor", "mixer", "vocabulary"
        ]
        domain_hits = sum(1 for t in domain_terms if t in a_lower)
        scores["domain_knowledge"] = min(domain_hits * 0.12, 1.0)

        # 8. Actionability — does it tell user what to DO?
        action_signals = [
            r'`[a-z_]+\s',          # Command syntax
            r'\brun\b|\buse\b|\btry\b|\btype\b',
            r'POST /api|GET /api',  # API endpoints
            r'python \w+',          # Script calls
        ]
        action_hits = sum(1 for p in action_signals if re.search(p, answer, re.I))
        scores["actionability"] = min(action_hits * 0.3, 1.0)

        # 9. Completeness — answer doesn't trail off
        last_char = answer.strip()[-1] if answer.strip() else ''
        scores["completeness"] = 1.0 if last_char in '.!?)' else 0.7

        # ── Weighted final ────────────────────────────────────────────────────
        weights = {
            "relevance":        0.25,
            "length":           0.12,
            "specificity":      0.15,
            "honesty":          0.12,
            "structure":        0.12,
            "unknown_handling": 0.04,
            "domain_knowledge": 0.10,
            "actionability":    0.06,
            "completeness":     0.04,
        }
        final = sum(scores[k] * weights[k] for k in weights)

        return {
            "total": round(final, 3),
            "breakdown": {k: round(v, 3) for k, v in scores.items()},
            "grade": self._grade(final),
        }

    def _grade(self, score: float) -> str:
        if score >= 0.85: return "A+ (Excellent)"
        if score >= 0.75: return "A  (Good)"
        if score >= 0.65: return "B  (Acceptable)"
        if score >= 0.50: return "C  (Needs improvement)"
        return                    "D  (Poor — retrain)"

    def _tokens(self, text: str) -> List[str]:
        return re.findall(r'\b[a-z]{3,}\b', text.lower())

# STEP 3: CONSTITUTIONAL AI CRITIQUE
# =============================================================================

def cai_critique(question: str, answer: str) -> Dict:
    """
    Constitutional AI: critique the answer against each rule.
    Returns violations and a revised answer suggestion.
    
    This is exactly how Claude refines its own outputs.
    """
    violations = []
    suggestions = []

    for rule in CONSTITUTION:
        violated = False

        # Rule: Must directly address the question
        if "directly address" in rule:
            q_words = set(re.findall(r'\b\w{3,}\b', question.lower()))
            a_words = set(re.findall(r'\b\w{3,}\b', answer.lower()))
            # Short questions (< 5 words) naturally have low overlap — skip check
            if len(q_words) >= 5 and len(q_words & a_words) < 2:
                violated = True
                suggestions.append("Include more words from the question in your answer.")

        # Rule: Must not be vague
        elif "vague" in rule:
            vague_phrases = ["it depends", "various things", "many ways", "generally speaking",
                             "could be many", "there are multiple"]
            if any(p in answer.lower() for p in vague_phrases) and len(answer.split()) < 30:
                violated = True
                suggestions.append("Be more specific — avoid vague phrases without follow-up detail.")

        # Rule: If AI doesn't know, must say so
        elif "doesn't know" in rule:
            knowledge_claim = re.search(r'\b(is|are|was|the answer is|means)\b', answer.lower())
            word_count = len(answer.split())
            if knowledge_claim and word_count < 20:
                violated = True
                suggestions.append("If unsure, hedge with 'I think' or 'based on my knowledge'.")

        # Rule: Must give examples (only for factual/technical questions)
        elif "concrete examples" in rule:
            is_technical = any(w in question.lower() for w in
                               ["how", "what is", "explain", "algorithm", "work", "does"])
            has_example = bool(re.search(r'e\.g\.|for example|such as|like |for instance|\bexample\b', answer, re.I))
            has_numbered = bool(re.search(r'^\d+\.', answer, re.M))
            has_bullet   = bool(re.search(r'^[-•\*]', answer, re.M))
            if is_technical and not has_example and not has_numbered and not has_bullet and len(answer.split()) > 40:
                violated = True
                suggestions.append("Add a concrete example to illustrate the concept.")

        # Rule: No made-up facts
        elif "make up facts" in rule:
            overconfident = re.search(r'\b(definitely|certainly|guaranteed|always|never|100%)\b', answer, re.I)
            if overconfident and len(answer.split()) < 50:
                violated = True
                suggestions.append("Soften strong claims — use 'likely', 'I believe', etc.")

        # Rule: Compression must mention algorithms
        elif "compression must mention" in rule:
            if any(w in question.lower() for w in ["compress", "algorithm", "shrink", "zip"]):
                algo_mentioned = any(w in answer.lower() for w in
                                     ["cmix", "bwt", "lz77", "ppm", "huffman", "arithmetic", "myzip"])
                if not algo_mentioned:
                    violated = True
                    suggestions.append("Mention specific compression algorithms (CMIX, BWT, LZ77, PPM).")

        if violated:
            violations.append(rule)

    passed   = len(CONSTITUTION) - len(violations)
    cai_score = passed / len(CONSTITUTION)

    return {
        "violations":   violations,
        "violations_count": len(violations),
        "rules_passed": passed,
        "total_rules":  len(CONSTITUTION),
        "cai_score":    round(cai_score, 3),
        "suggestions":  suggestions,
        "passed": len(violations) == 0,
    }

# AUTO-EXTRACT PAIRS FROM ANTHROPIC LEAK FILES

def generate_pairs_from_anthropic(anthropic_dir: Path, max_pairs: int = 30) -> List[Dict]:
    """
    Auto-extract Q&A style training pairs from Anthropic leaked system prompts.
    Looks for: example blocks, rationale blocks, user/response patterns.
    Returns list of SFT pairs ready to feed into run_sft().
    """
    pairs = []
    md_files = sorted(anthropic_dir.glob("*.md"))

    for md_file in md_files:
        text = md_file.read_text(encoding="utf-8", errors="replace")
        fname = md_file.stem

        # Pattern 1: `<user>` ... `</user>` + `<response>` ... `</response>`
        pattern = re.compile(
            r'`<user>`\s*([\s\S]+?)`</user>`[\s\S]*?`<(?:good_response|response)>`\s*([\s\S]+?)`</(?:good_response|response)>`',
            re.MULTILINE
        )
        for m in pattern.finditer(text):
            q = m.group(1).strip()[:300]
            a = m.group(2).strip()[:1000]
            if len(q) > 10 and len(a) > 20:
                # Skip action-only answers ([searches the web...])
                if not re.match(r'^\[.*\]$', a.strip()):
                    pairs.append({
                        "question": q,
                        "ideal_answer": a,
                        "topic": f"anthropic example from {fname}",
                        "skills": ["leaked Anthropic training pattern", "real example"]
                    })

        # Pattern 2: User: "..." → Assistant: "..."
        pattern2 = re.compile(
            r'User:\s*"([^"]+)"\s*\n+Assistant:\s*([\s\S]+?)(?=\nUser:|\Z)',
            re.MULTILINE
        )
        for m in pattern2.finditer(text):
            q = m.group(1).strip()[:300]
            a = m.group(2).strip()[:1000]
            if len(q) > 10 and len(a) > 30:
                pairs.append({
                    "question": q,
                    "ideal_answer": a,
                    "topic": f"anthropic pattern from {fname}",
                    "skills": ["Anthropic conversation style", "real training pair"]
                })

        if len(pairs) >= max_pairs:
            break

    return pairs[:max_pairs]

def load_feedback() -> List[Dict]:
    if FEEDBACK_FILE.exists():
        return json.loads(FEEDBACK_FILE.read_text())
    return []

def save_feedback(data: List[Dict]):
    FEEDBACK_FILE.write_text(json.dumps(data, indent=2, ensure_ascii=False))

def record_feedback(question: str, answer: str, rating: float, comment: str = ""):
    """
    Record human feedback on an AI answer.
    rating: 1.0 = perfect, 0.5 = ok, 0.0 = bad
    This is the RLHF signal — the equivalent of Anthropic's human labelers.
    """
    feedback = load_feedback()
    rm = RewardModel()
    auto_score = rm.score(question, answer)
    cai_result = cai_critique(question, answer)

    entry = {
        "id":          len(feedback) + 1,
        "timestamp":   time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
        "question":    question,
        "answer":      answer,
        "human_rating": rating,
        "auto_score":  auto_score["total"],
        "cai_score":   cai_result["cai_score"],
        "combined":    round((rating * 0.5 + auto_score["total"] * 0.3 + cai_result["cai_score"] * 0.2), 3),
        "comment":     comment,
        "violations":  cai_result["violations"],
    }
    feedback.append(entry)
    save_feedback(feedback)
    return entry

def analyze_feedback_patterns() -> Dict:
    """Analyze all feedback to find systematic weaknesses."""
    feedback = load_feedback()
    if not feedback:
        return {"error": "No feedback recorded yet"}

    low_rated  = [f for f in feedback if f["human_rating"] < 0.5]
    high_rated = [f for f in feedback if f["human_rating"] >= 0.8]
    
    # Find which topics get bad ratings
    bad_topics = defaultdict(int)
    for f in low_rated:
        words = re.findall(r'\b\w{4,}\b', f["question"].lower())
        for w in words[:5]:
            bad_topics[w] += 1

    # Find common violations
    violation_counts = defaultdict(int)
    for f in feedback:
        for v in f.get("violations", []):
            violation_counts[v[:50]] += 1

    avg_human  = sum(f["human_rating"] for f in feedback) / len(feedback)
    avg_auto   = sum(f["auto_score"] for f in feedback) / len(feedback)

    return {
        "total_feedback":     len(feedback),
        "low_rated_count":    len(low_rated),
        "high_rated_count":   len(high_rated),
        "avg_human_rating":   round(avg_human, 3),
        "avg_auto_score":     round(avg_auto, 3),
        "weak_topics":        dict(sorted(bad_topics.items(), key=lambda x: -x[1])[:10]),
        "top_violations":     dict(sorted(violation_counts.items(), key=lambda x: -x[1])[:5]),
        "agreement_rate":     round(sum(1 for f in feedback if abs(f["human_rating"] - f["auto_score"]) < 0.2) / len(feedback), 3),
    }

# STEP 6: SFT TRAINER  — Feed ideal pairs into the brain
# =============================================================================

def run_sft(pairs: List[Dict] = None, verbose: bool = True) -> Dict:
    """
    Supervised Fine-Tuning: store ideal (question→answer) pairs in the brain.
    The brain learns by example — seeing thousands of good answers makes it better.
    """
    if pairs is None:
        pairs = SFT_PAIRS

    rm    = RewardModel()
    stats = {"learned": 0, "skipped": 0, "avg_score": 0.0, "scores": []}

    if verbose:
        print(f"\n🎓 SFT Phase — Loading {len(pairs)} training pairs into brain...")
        print("  (This teaches the AI what a GOOD answer looks like)\n")

    for i, pair in enumerate(pairs, 1):
        q   = pair["question"]
        a   = pair["ideal_answer"]
        topic = f"training: {pair.get('topic', 'general')}"

        # Score the ideal answer (should be high)
        score = rm.score(q, a)
        stats["scores"].append(score["total"])

        # Store as knowledge: format = "Q: ... A: ..."
        content = f"Question: {q}\n\nIdeal Answer: {a}\n\nSkills demonstrated: {', '.join(pair.get('skills', []))}"
        # Sanitize topic — remove chars that are invalid in Windows filenames
        safe_topic = re.sub(r'[<>:"/\\|?*]', '-', topic)

        try:
            result = brain.store_knowledge(
                topic=safe_topic,
                content=content,
                source="sft_training",
                source_url=""
            )
            stats["learned"] += 1
            if verbose:
                print(f"  [{i:2d}] ✅ {topic:<40} reward={score['total']:.2f} ({score['grade']})")
        except Exception as e:
            stats["skipped"] += 1
            if verbose:
                print(f"  [{i:2d}] ❌ {topic} — {e}")

    stats["avg_score"] = round(sum(stats["scores"]) / max(len(stats["scores"]), 1), 3)
    stats.pop("scores")

    # Save SFT record
    existing = json.loads(SFT_FILE.read_text()) if SFT_FILE.exists() else []
    existing.append({"timestamp": time.strftime("%Y-%m-%dT%H:%M:%SZ"), **stats})
    SFT_FILE.write_text(json.dumps(existing, indent=2))

    if verbose:
        print(f"\n  ✅ SFT Complete: {stats['learned']} pairs learned, avg reward={stats['avg_score']:.2f}")
    return stats

# STEP 7: CAI SELF-IMPROVEMENT LOOP
# =============================================================================

def run_cai_loop(verbose: bool = True) -> Dict:
    """
    Constitutional AI: load brain's stored Q&A pairs,
    critique each against the constitution, and flag violations.
    Violations become new training targets.
    """
    index = brain.load_index()
    entries = list(index.get("entries", {}).values())

    # Focus on entries that look like answers (have Q&A format)
    qa_entries = [e for e in entries if "question" in e.get("summary", "").lower()
                  or e.get("source") == "sft_training"]

    rm    = RewardModel()
    critiques = []
    improved  = 0
    violations_found = 0

    if verbose:
        print(f"\n🔍 CAI Phase — Critiquing {len(qa_entries)} knowledge entries...")
        print("  (The AI checks its own answers against the Constitution)\n")

    for entry in qa_entries[:50]:  # Cap at 50 to avoid too long
        raw_file = entry.get("raw_file", "")
        if not os.path.exists(raw_file):
            continue
        
        with open(raw_file, "r", encoding="utf-8") as f:
            content = f.read()

        # Extract Q and A from content
        q_match = re.search(r'Question:\s*(.+?)(?:\n|$)', content)
        a_match = re.search(r'Ideal Answer:\s*([\s\S]+?)(?:Skills demonstrated:|$)', content)

        if not (q_match and a_match):
            continue

        q = q_match.group(1).strip()
        a = a_match.group(1).strip()

        cai = cai_critique(q, a)
        reward = rm.score(q, a)

        if cai["violations"]:
            violations_found += len(cai["violations"])
            critiques.append({
                "entry_id": entry["id"],
                "question": q,
                "violations": cai["violations"],
                "suggestions": cai["suggestions"],
                "cai_score": cai["cai_score"],
                "reward_score": reward["total"],
            })

            if verbose:
                print(f"  ⚠️  {entry['topic'][:50]}")
                for s in cai["suggestions"][:2]:
                    print(f"       → {s}")
        else:
            improved += 1
            if verbose:
                print(f"  ✅  {entry['topic'][:50]}  (passes all {len(CONSTITUTION)} rules)")

    # Save critiques for future improvement
    CAI_FILE.write_text(json.dumps(critiques, indent=2, ensure_ascii=False))

    result = {
        "entries_checked": len(qa_entries),
        "passed_all_rules": improved,
        "violations_found": violations_found,
        "entries_with_violations": len(critiques),
    }

    if verbose:
        print(f"\n  CAI Results:")
        print(f"  ✅ Passed: {improved}")
        print(f"  ⚠️  Violations: {violations_found} across {len(critiques)} entries")

    return result

        parts = args.score.split("|", 1)
        if len(parts) == 2:
            q, a = parts
            rm = RewardModel()
            score = rm.score(q.strip(), a.strip())
            cai   = cai_critique(q.strip(), a.strip())
            print(f"Question : {q.strip()}")
            print(f"Answer   : {a.strip()[:100]}...")
            print(f"\nReward Score: {score['total']} ({score['grade']})")
            print(f"Breakdown   : {score['breakdown']}")
            print(f"\nCAI Score   : {cai['cai_score']} ({cai['rules_passed']}/{cai['total_rules']} rules passed)")
            if cai["suggestions"]:
                print("Suggestions :")
                for s in cai["suggestions"]:
                    print(f"  → {s}")
        return

    if args.sft or args.full:
        print("\n" + "="*60)
        print("PHASE 1: Supervised Fine-Tuning (SFT)")
        print("="*60)
        print(f"Teaching the AI {len(SFT_PAIRS)} ideal (question → answer) examples...")
        print("This is how Anthropic's human contractors teach Claude.")
        sft_stats = run_sft()
        print()

    if args.anthropic or args.full:
        anthropic_dir = BASE_DIR / "Anthropic"
        if anthropic_dir.exists():
            print("\n" + "="*60)
            print("PHASE 1b: Auto-Extract from Anthropic Leak Files")
            print("="*60)
            print("Mining real Q&A training pairs from leaked Claude system prompts...")
            extracted = generate_pairs_from_anthropic(anthropic_dir, max_pairs=40)
            if extracted:
                print(f"  → Extracted {len(extracted)} real Anthropic example pairs")
                run_sft(pairs=extracted)
            else:
                print("  → No structured Q&A pairs found in the files")
            print()

    if args.cai or args.full:
        print("\n" + "="*60)
        print("PHASE 2: Constitutional AI (CAI) Critique")
        print("="*60)
        print("The AI is checking its own stored knowledge against the Constitution...")
        cai_stats = run_cai_loop()
        print()

    if args.feedback:
        print("\n" + "="*60)
        print("RLHF Feedback Analysis")
        print("="*60)
        patterns = analyze_feedback_patterns()
        if "error" in patterns:
            print(f"  {patterns['error']}")
        else:
            print(f"  Total feedback    : {patterns['total_feedback']}")
            print(f"  Avg human rating  : {patterns['avg_human_rating']}")
            print(f"  Avg auto score    : {patterns['avg_auto_score']}")
            print(f"  Agreement rate    : {patterns['agreement_rate']:.0%}")
            print(f"  Weak topics       : {patterns['weak_topics']}")
            print(f"  Top violations    : {patterns['top_violations']}")
        print()

    if args.full or args.sft or args.cai:
        print("="*60)
        print("✅ TRAINING CYCLE COMPLETE")
        print("="*60)
        print()
        print_status()

    print("""

   5. Expand the CONSTITUTION with your own rules
""")
